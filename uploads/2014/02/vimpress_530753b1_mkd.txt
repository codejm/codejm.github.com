
install redis
1. download redis package http://download.redis.io/releases/redis-2.8.6.tar.gz
2. tar zxvf redis-2.8.6.tar.gz
3. cd redis-2.8.6/
4. sudo chmod 755 ./ -R
5. cd utils
6. ./install_server


install redis-py
1. git clone https://github.com/andymccurdy/redis-py.git
2. cd redis-py
3. sudo chmod 755 ./ -R
4. sudo python setup.py install
5. test: import redis
